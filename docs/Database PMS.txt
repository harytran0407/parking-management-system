-- ======================================================
-- NHÓM 1: QUẢN LÝ NGƯỜI DÙNG & PHÂN QUYỀN
-- ======================================================

-- 1. Bảng Role (Phân quyền)
CREATE TABLE ROLE (
	ROLE_ID INT AUTO_INCREMENT PRIMARY KEY,
    ROLE_NAME VARCHAR(50) UNIQUE NOT NULL,
    DESCRIPTION VARCHAR(200)
);

-- 2. Bảng User (Tài khoản người dùng)
-- Bảng này đóng vai trò lưu trữ thông tin của đội ngũ vận hành (tương ứng với các primary actors như Parking Staff, Parking Manager, System Administrator), 
-- những người thực hiện các nghiệp vụ quản lý bãi xe 
CREATE TABLE USERS (
    USER_ID VARCHAR(4) PRIMARY KEY,
    USERNAME VARCHAR(50) UNIQUE NOT NULL,
    PASSWORD VARCHAR(50) NOT NULL,
    FULL_NAME VARCHAR(100),
    PHONE VARCHAR(15),
    EMAIL VARCHAR(100),
    STATUS ENUM('ACTIVE', 'INACTIVE', 'PENDING' , 'BANNED') DEFAULT 'ACTIVE',				-- Dùng ENUM thay vì Varchar hay Boolean để tránh nhập dữ liệu rác
    CREATE_AT DATETIME DEFAULT CURRENT_TIMESTAMP,
	ROLE_ID INT NOT NULL,
    
    CONSTRAINT FK_USER_ROLE FOREIGN KEY (ROLE_ID) REFERENCES ROLE(ROLE_ID)
);

-- ======================================================
-- NHÓM 2: QUẢN LÝ LOẠI PHƯƠNG TIỆN
-- ======================================================

-- 3. Bảng Vehicle_Type (Loại phương tiện)
CREATE TABLE VEHICLE_TYPE (
	VEHICLE_TYPE_ID INT AUTO_INCREMENT PRIMARY KEY,											-- Id của loại phương tiện
    VEHICLE_TYPE_NAME VARCHAR(100) NOT NULL,												-- Tên của loại phương tiện
    USER_ID VARCHAR (4),																	-- User của loại phương tiện
    
    CONSTRAINT FK_TYPE_USER FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)
    );

-- 4. Bảng Vehicle_Owner (Thông tin chủ xe)
CREATE TABLE VEHICLE_OWNER (
	VEHICLE_OWNER_ID INT AUTO_INCREMENT PRIMARY KEY,										-- Id của chủ phương tiện
    VEHICLE_OWNER_NAME VARCHAR(100) NOT NULL, 												-- Tên của chủ phương tiện
    VEHICLE_OWNER_PHONE VARCHAR(15) NOT NULL,												-- SĐT của chủ phương tiện
    VEHICLE_OWNER_EMAIL VARCHAR(250),														-- Email của chủ phương tiện
    OWNER_USERNAME VARCHAR(50) NOT NULL,													-- Tên đăng nhập
    OWNER_PASSWORD VARCHAR(50) NOT NULL, 													-- Mật Khẩu
    STATUS ENUM('ACTIVE', 'INACTIVE', 'BANNED', 'PENDING') DEFAULT 'ACTIVE',				-- Trạng thái 
    USER_ID VARCHAR(4),
    
    CONSTRAINT FK_OWNER_USER FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)
);

-- 5. Bảng Vehicle (Phương tiện)
-- Bảng này là nơi kết nối Chủ xe và Loại phương tiện
CREATE TABLE VEHICLE (
	VEHICLE_ID INT AUTO_INCREMENT PRIMARY KEY,												-- Id của phương tiện
    VEHICLE_PLATE_NUMBER VARCHAR(50) NOT NULL,												-- Biển số xe
    VEHICLE_DESCRIPTION VARCHAR(200),														-- Mô tả xe (xe điện hay xe xăng, màu sắc, kích thước)
	VEHICLE_TYPE_ID INT NOT NULL,															-- Khóa ngoại nối đến bảng Loại phương tiện
    VEHICLE_OWNER_ID INT NOT NULL,															-- Khóa ngoại nối đến bảng Chủ phương tiện
    
    CONSTRAINT FK_VEHICLE_TYPE FOREIGN KEY (VEHICLE_TYPE_ID) REFERENCES VEHICLE_TYPE(VEHICLE_TYPE_ID),
    CONSTRAINT FK_VEHICLE_OWNER FOREIGN KEY (VEHICLE_OWNER_ID) REFERENCES VEHICLE_OWNER(VEHICLE_OWNER_ID)
);

-- ======================================================
-- NHÓM 3: QUẢN LÝ BÃI ĐỖ, ĐẶT CHỖ & THANH TOÁN
-- ======================================================

-- 6. Bảng Floor_Zone (Khu vực / Tầng đỗ xe)
CREATE TABLE FLOOR_ZONE (
    ZONE_ID INT AUTO_INCREMENT PRIMARY KEY,                                					-- Id của Khu vực/Tầng
    ZONE_NAME VARCHAR(50) NOT NULL,                                        					-- Tên của tầng, khu vực, VD: Tầng 1, Khu A
    CAPACITY INT NOT NULL,                                                  				-- Sức chứa tối đa
    STATUS ENUM('ACTIVE', 'MAINTENANCE') DEFAULT 'ACTIVE',                  				-- Trạng thái khu vực
	VEHICLE_TYPE_ID INT NOT NULL,                                          					-- Khu vực này dành cho loại xe nào
    
    CONSTRAINT FK_ZONE_TYPE FOREIGN KEY (VEHICLE_TYPE_ID) REFERENCES VEHICLE_TYPE(VEHICLE_TYPE_ID)
);

-- 7. Bảng Parking_Slot (Vị trí đỗ xe chi tiết)
CREATE TABLE PARKING_SLOT (
    SLOT_ID INT AUTO_INCREMENT PRIMARY KEY,                                 				-- Id của vị trí đỗ
    SLOT_NAME VARCHAR(20) NOT NULL,                                         				-- VD: A-15, B-02
    STATUS ENUM('AVAILABLE', 'OCCUPIED', 'RESERVED', 'MAINTENANCE') DEFAULT 'AVAILABLE',	-- Trạng thái
	ZONE_ID INT NOT NULL,                                                  					-- Thuộc khu vực nào
	USER_ID VARCHAR(4),                                 									-- Nhân viên cập nhật trạng thái

    CONSTRAINT FK_SLOT_ZONE FOREIGN KEY (ZONE_ID) REFERENCES FLOOR_ZONE(ZONE_ID),
	CONSTRAINT FK_SLOT_USER FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)
);

-- 8. Bảng Booking (Đặt chỗ trước)
CREATE TABLE BOOKING (
    BOOKING_ID INT AUTO_INCREMENT PRIMARY KEY,          									
    DURATION INT,                            												-- Thời lượng đặt chỗ
	EXPECTED_ARRIVAL DATETIME NOT NULL,                 									-- Giờ dự kiến xe sẽ đến bãi
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP,      									-- Thời gian tạo đơn đặt chỗ
    BOOKING_STATUS ENUM('PROCESSING', 'APPROVED', 'CANCELLED', 'REJECTED', 'COMPLETED') DEFAULT 'PROCESSING',
    REMARKS VARCHAR(255),                               									-- Ghi chú bổ sung cho giao dịch
	VEHICLE_OWNER_ID INT NOT NULL,                      									-- Khách hàng thực hiện đặt chỗ
	VEHICLE_ID INT NOT NULL,                            									-- Khóa ngoại nối đến thông tin chiếc xe
	SLOT_ID INT,                                        									-- Khóa ngoại nối đến vị trí đỗ được chỉ định
	USER_ID VARCHAR(4),                                 									-- Nhân viên xử lý giao dịch đặt chỗ
    
    -- Các ràng buộc khóa ngoại (Foreign Keys)
    CONSTRAINT FK_BOOKING_OWNER FOREIGN KEY (VEHICLE_OWNER_ID) REFERENCES VEHICLE_OWNER(VEHICLE_OWNER_ID),
    CONSTRAINT FK_BOOKING_VEHICLE FOREIGN KEY (VEHICLE_ID) REFERENCES VEHICLE(VEHICLE_ID),
    CONSTRAINT FK_BOOKING_SLOT FOREIGN KEY (SLOT_ID) REFERENCES PARKING_SLOT(SLOT_ID),
    CONSTRAINT FK_BOOKING_USER FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)
);

-- 9. Bảng Parking_Session (Phiên đỗ xe)
-- Ghi lại mỗi lượt xe chạy vào và chạy ra khỏi bãi
CREATE TABLE PARKING_SESSION (
    SESSION_ID INT AUTO_INCREMENT PRIMARY KEY,                              				-- Id phiên đỗ
    LICENSE_PLATE_IN VARCHAR(50) NOT NULL,                                  				-- Biển số chụp được lúc vào (có thể khác với biển đăng ký)
    CHECK_IN_TIME DATETIME DEFAULT CURRENT_TIMESTAMP,                       				-- Giờ xe vào
    CHECK_OUT_TIME DATETIME,                                                				-- Giờ xe ra
	GATE_IN VARCHAR(50),                                                    				-- Cổng xe chạy vào
    GATE_OUT VARCHAR(50),                                                   				-- Cổng xe chạy ra
    STATUS ENUM('ACTIVE', 'COMPLETED') DEFAULT 'ACTIVE',                    				-- ACTIVE là đang đỗ, COMPLETED là đã ra
	VEHICLE_TYPE_ID INT NOT NULL,                                           				-- Loại xe lúc vào
    SLOT_ID INT,                                                            				-- Vị trí đỗ (có thể Null nếu đỗ tự do)
	VEHICLE_ID INT,                                                         				-- NULL nếu là khách vãng lai, CÓ VALUE nếu là khách đã đăng ký
	BOOKING_ID INT,                                                         				-- NULL nếu đỗ tự do, CÓ VALUE nếu xe vào theo lịch đặt trước
    STAFF_IN_ID VARCHAR(4),                                                 				-- NV trực lúc vào
    STAFF_OUT_ID VARCHAR(4),                                                				-- NV trực lúc ra
    
    CONSTRAINT FK_SESSION_TYPE FOREIGN KEY (VEHICLE_TYPE_ID) REFERENCES VEHICLE_TYPE(VEHICLE_TYPE_ID),
    CONSTRAINT FK_SESSION_SLOT FOREIGN KEY (SLOT_ID) REFERENCES PARKING_SLOT(SLOT_ID),
	CONSTRAINT FK_SESSION_VEHICLE FOREIGN KEY (VEHICLE_ID) REFERENCES VEHICLE(VEHICLE_ID),
	CONSTRAINT FK_SESSION_BOOKING FOREIGN KEY (BOOKING_ID) REFERENCES BOOKING(BOOKING_ID),
	CONSTRAINT FK_SESSION_STAFF_IN FOREIGN KEY (STAFF_IN_ID) REFERENCES USERS(USER_ID),
    CONSTRAINT FK_SESSION_STAFF_OUT FOREIGN KEY (STAFF_OUT_ID) REFERENCES USERS(USER_ID)
);

-- 10. Bảng Pricing_Policy (Bảng giá gửi xe)
CREATE TABLE PRICING_POLICY (
    POLICY_ID INT AUTO_INCREMENT PRIMARY KEY,                               				-- Id của bảng giá
    BASE_PRICE DECIMAL(10, 2) NOT NULL,                                     				-- Giá cơ bản (VD: block 4 giờ đầu)
    HOURLY_RATE DECIMAL(10, 2) NOT NULL,                                    				-- Phí phụ thu mỗi giờ tiếp theo
    OVERNIGHT_FEE DECIMAL(10, 2) NOT NULL,                                  				-- Phí phụ thu qua đêm
    EFFECTIVE_DATE DATE NOT NULL,                                           				-- Ngày bắt đầu áp dụng
	VEHICLE_TYPE_ID INT NOT NULL,                                           				-- Bảng giá áp dụng cho loại xe nào
    
    CONSTRAINT FK_POLICY_TYPE FOREIGN KEY (VEHICLE_TYPE_ID) REFERENCES VEHICLE_TYPE(VEHICLE_TYPE_ID)
);

-- 11. Bảng Monthly_Pass (Thẻ gửi xe tháng)
CREATE TABLE MONTHLY_PASS (
    CARD_ID INT AUTO_INCREMENT PRIMARY KEY,                                 				-- Id của thẻ tháng
    RFID_CODE VARCHAR(50) UNIQUE NOT NULL,                                  				-- Mã thẻ cứng để quẹt
    START_DATE DATE NOT NULL,                                               				-- Ngày bắt đầu có hiệu lực
    END_DATE DATE NOT NULL,                                                 				-- Ngày hết hạn
    STATUS ENUM('ACTIVE', 'EXPIRED', 'SUSPENDED') DEFAULT 'ACTIVE',         				-- Trạng thái thẻ
	VEHICLE_ID INT NOT NULL,                                                				-- Thẻ cấp cho xe nào
    
    CONSTRAINT FK_PASS_VEHICLE FOREIGN KEY (VEHICLE_ID) REFERENCES VEHICLE(VEHICLE_ID)
);

-- 12. Bảng Payment (Thanh toán phí)
CREATE TABLE PAYMENT (
    PAYMENT_ID INT AUTO_INCREMENT PRIMARY KEY,          									-- Id giao dịch thanh toán
    PAYMENT_TYPE ENUM('SESSION', 'MONTHLY_PASS', 'BOOKING') NOT NULL,       				-- Giúp phân loại giao dịch khi làm báo cáo doanh thu
    AMOUNT_DUE DECIMAL(10, 2) NOT NULL,                 									-- Tổng tiền phải trả
    AMOUNT_PAID DECIMAL(10, 2) NOT NULL,                									-- Tiền khách đã trả
    PAYMENT_METHOD ENUM('CASH', 'VNPAY', 'MOMO', 'BANK_TRANSFER') NOT NULL, 				-- Tiền mặt chỉ dùng tại quầy, VNPAY/MOMO dùng trên web
    PAYMENT_TIME DATETIME DEFAULT CURRENT_TIMESTAMP,                        
    STATUS ENUM('PENDING', 'SUCCESS', 'FAILED', 'REFUNDED') DEFAULT 'SUCCESS', 				-- Trạng thái giao dịch mạng (Thành công, lỗi, đang xử lý hoặc hoàn tiền)
    PAYMENT_STATUS ENUM('UNPAID', 'PARTIAL', 'PAID') DEFAULT 'UNPAID', 						-- Trạng thái thanh toán (Chưa trả, Trả một phần, Đã trả)
	REMARKS VARCHAR(255),                                                                   -- Ghi chú lý do hoàn tiền/ lỗi giao dịch
    PAID_BY INT,                                        									-- Người thanh toán (có thể NULL nếu khách vãng lai)
    SESSION_ID INT,                                                         				-- Trả tiền cho lượt đỗ thực tế tại cổng ra
    CARD_ID INT,                                                            				-- Trả tiền khi mua/gia hạn thẻ tháng
    BOOKING_ID INT,                                                         				-- Trả tiền cọc/phí giữ chỗ trên web
    USER_ID VARCHAR(4),                                 									-- Nhân viên thu tiền / xử lý giao dịch

    CONSTRAINT FK_PAY_SESSION FOREIGN KEY (SESSION_ID) REFERENCES PARKING_SESSION(SESSION_ID),
    CONSTRAINT FK_PAY_CARD FOREIGN KEY (CARD_ID) REFERENCES MONTHLY_PASS(CARD_ID),
    CONSTRAINT FK_PAY_BOOKING FOREIGN KEY (BOOKING_ID) REFERENCES BOOKING(BOOKING_ID),
    CONSTRAINT FK_PAY_PAID_BY FOREIGN KEY (PAID_BY) REFERENCES VEHICLE_OWNER(VEHICLE_OWNER_ID),
    CONSTRAINT FK_PAY_USER FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)
);

-- ======================================================

-- ======================================================

-- 13. Bảng Incident_Log (Biên bản ghi nhận sự cố)
CREATE TABLE INCIDENT_LOG (
    LOG_ID INT AUTO_INCREMENT PRIMARY KEY,                                  				-- Id sự cố
    SESSION_ID INT NOT NULL,                                                				-- Liên quan đến phiên đỗ nào
    REPORTED_BY VARCHAR(4) NOT NULL,                                        				-- Nhân viên nào báo cáo
    ISSUE_TYPE ENUM('LOST_TICKET', 'WRONG_PLATE', 'WRONG_SLOT', 'SYSTEM_ERROR', 'OTHER') NOT NULL,
    DESCRIPTION VARCHAR(500),                                               				-- Mô tả chi tiết sự cố
    REPORT_TIME DATETIME DEFAULT CURRENT_TIMESTAMP,                         				-- Thời gian ghi nhận
    STATUS ENUM('OPEN', 'RESOLVED') DEFAULT 'OPEN',                         				-- OPEN: Đang xử lý, RESOLVED: Đã giải quyết
    
    CONSTRAINT FK_INCIDENT_SESSION FOREIGN KEY (SESSION_ID) REFERENCES PARKING_SESSION(SESSION_ID),
    CONSTRAINT FK_INCIDENT_STAFF FOREIGN KEY (REPORTED_BY) REFERENCES USERS(USER_ID)
);
